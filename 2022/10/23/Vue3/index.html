Vue学习笔记

1、创建Vue项目

使用工具：Node.js、Idea_Profession、Vue3

详戳 Idea创建vue项目

Vue3官方文档： 简介 | Vue.js (vuejs.org)

2、Vue的使用

初步使用

<script>
  export default {
    data(){
      return{
        name: '司文轩',
        age: 56
      }
    }
  }
</script>
<template>
  <div id="app">
    <p>{{name}}</p>
    <p>{{age}}</p>
  </div>
</template>

按钮

<template>
  <div id="app">
    <p>{{name}}</p>
    <p>{{age}}</p>
    <!-- <button @click="方法名">按钮名</button> -->
    <button @click="changeName">改名字</button>
  </div>
</template>

方法

<script>
  export default {
    data() {
      return {
        name: '司文轩',
        age: 56
      }
    },
    methods: {
      // 给Vue定义方法
      changeName: function () {
        // this指向Vue实例
        this.name = '王八蛋'
      }
    }
  }
</script>

<template>
  <div id="app">
    <p>{{name}}</p>
    <p>{{age}}</p>
    <button @click="changeName">改名字</button>
  </div>
</template>

动态修改页面元素

v-html

<script>
  export default {
    data() {
      return {
        mes: '<h1>大标题<h1>',
      };
    },
  };

</script>
<template>
  <div id="app">
    <p>{{mes}}</p>
    <p v-html="mes"></p>
  </div>
</template>
