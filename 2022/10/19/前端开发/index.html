sort()

需要借助函数实现

// sort()当 a > b 时，需要交换则返回正数，不需要交换返回负数， a < b 时亦然

// 直接改变数组指向的地址

arr.sort(function (a,b){
    return a - b
});

> [ 1, 3, 4, 5, 9 ]

// 降序
var arr = [3,5,9,4,1];
arr.sort(function (a,b){
    return b-a;
});
console.log(arr);
[ 9, 5, 4, 3, 1 ]

深克隆

var arr1 = [1,2,3,4,[5,6]];
// 使用函数的递归
function deepClone(arr){
    // 创建储存结果的数组
    var result = [];
    for(var i = 0; i < arr.length; i++){
        if(Array.isArray(arr[i])){
            // 若该项为数组，则再次调用该函数
            result.push(deepClone(arr[i]));
        }else{
            // 若为数值，则将值推入结果数组中
            result.push(arr[i]);
        }
    }
    // 返回结果数组
    return result;
}
var arr2 = deepClone(arr1);
console.log(arr2);
> Array(5) [ 1, 2, 3, 4, (2) […] ]
> 0: 1
> 1: 2
> 2: 3
> 3: 4
> 4: Array [ 5, 6 ]
console.log(arr2 == arr1)
> false

DOM

基本概念

​ DOM是JS操控HTML和CSS的桥梁。

​ 文档对象模型 (Document Object Model) 是 HTML 和 XML 文档的编程接口。它提供了对文档的结构化的表述，并定义了一种方式可以使从程序中对该结构进行访问，从而改变文档的结构，样式和内容。DOM 将文档解析为一个由节点和对象（包含属性和方法的对象）组成的结构集合。简言之，它会将 web 页面和脚本或程序语言连接起来。

<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <style>
        td{
            width: 20px;
            height: 20px;
            border: 1px solid #000;
        }
    </style>

<body>
    <table id="myTable"></table>
    <script>
        var myTable = document.getElementById('myTable');
        // 创建20行
        for(var i = 0; i < 20; i++){
            var tr = document.createElement('tr');
            // 创建12列
            for(var j = 0; j < 12; j++){
                var td = document.createElement('td');
                // 让 tr 追加 td 标签
                tr.appendChild(td);
            }
            // 让 myTable 追加 tr 标签
            myTable.appendChild(tr);
        }
    </script>
</body>

常见的节点关系函数

<body>
    <div id="box1">
        <p>我只是个段落</p>
        <p>我是段落1</p>
        <p id="para">我是段落2</p>
        <p>我是段落3</p>
        <p>我不是个段落，才怪</p>
    </div>
    <script>
        var box1 = document.getElementById('box1');
        var para = document.getElementById('para');
        
        // ----------------------------------------------------------------------
        // IE6 能兼容的“寻找所有元素子节点”函数
        function getChildren(node){
            var children = [];
            // 寻找所有子节点里nodeType为1的节点，为所有元素节点，推入children
            for(var i = 0; i < node.childNodes.length; i++){
                if(node.childNodes[i].nodeType == 1){
                    children.push(node.childNodes[i]);
                }
            }
            return children;
        }
        console.log(getChildren(box1));
        
        // ----------------------------------------------------------------------
        // IE6 能兼容的“寻找前一个元素兄弟节点”函数
        function getElementPreviousSibling(node){
            var o = node;
            // 获取node的前一个节点，若为元素节点，则return，反之继续往前寻找
            while(o.previousSibling != null){
                if(o.previousSibling.nodeType == 1){
                    return o.previousSibling;
                }else{
                    o = o.previousSibling;
                }
            }
            // 若无前一个结点则返回null
            return null;
        }
        console.log(getElementPreviousSibling(para));
        
        // ----------------------------------------------------------------------
        // 获得某元素的所有兄弟节点
        // 方法一（此处使用）：遍历前面和后面兄弟节点，若为元素节点，前面的从前压入，本身和后面的从后推入
        // 方法二：遍历父节点的子节点，将元素节点推入结果中
        function getAllElementSibling(node){
            var sibling = [];
            var pre = node, next = node;
            // 前面的兄弟节点
            while(pre.previousSibling != null){
                if (pre.previousSibling.nodeType == 1){
                    sibling.unshift(pre.previousSibling);
                }
                pre = pre.previousSibling;
            }
            // 自身节点判断
            if(node.nodeType == 1){
                sibling.push(node);
            }
            //后面的兄弟节点
            while(next.nextSibling != null){
                if(next.nextSibling.nodeType == 1){
                    sibling.push(next.nextSibling);
                }
                next = next.nextSibling;
            }
            return sibling;
        }
        console.log(getAllElementSibling(para));
    </script>
