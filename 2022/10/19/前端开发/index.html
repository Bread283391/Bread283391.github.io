splice()

sort()

需要借助函数实现

// sort()当 a > b 时，需要交换则返回正数，不需要交换返回负数， a < b 时亦然
// a > b ，a - b > 0，所以 a 与 b 发生交换，结果为 ba ，为从小到大升序排序
// 直接改变数组指向的地址
//升序
var arr = [3,5,9,4,1];
arr.sort(function (a,b){
    return a - b
});
console.log(arr);
> [ 1, 3, 4, 5, 9 ]

// 降序
var arr = [3,5,9,4,1];
arr.sort(function (a,b){
    return b-a;
});
console.log(arr);
[ 9, 5, 4, 3, 1 ]

克隆

浅克隆
// 克隆的结果不再是指向同一个地址了，而是两个地址
var arr1 = [1,2,3,4,5];
var arr2 = [];
for (var i = 0; i < arr1.length; i++){
    arr2[i] = arr1 [i];
}
console.log(arr2);
console.log(arr1 == arr2);
> [ 1, 2, 3, 4, 5 ]
> false
深克隆
var arr1 = [1,2,3,4,[5,6]];
// 使用函数的递归
function deepClone(arr){
    // 创建储存结果的数组
    var result = [];
    for(var i = 0; i < arr.length; i++){
        if(Array.isArray(arr[i])){
            // 若该项为数组，则再次调用该函数
            result.push(deepClone(arr[i]));
        }else{
            // 若为数值，则将值推入结果数组中
            result.push(arr[i]);
        }
    }
    // 返回结果数组
    return result;
}
var arr2 = deepClone(arr1);
console.log(arr2);
> Array(5) [ 1, 2, 3, 4, (2) […] ]
> 0: 1
> 1: 2
> 2: 3
> 3: 4
> 4: Array [ 5, 6 ]
console.log(arr2 == arr1)
> false

DOM

基本概念

​ DOM是JS操控HTML和CSS的桥梁。

​ 文档对象模型 (Document Object Model) 是 HTML 和 XML 文档的编程接口。它提供了对文档的结构化的表述，并定义了一种方式可以使从程序中对该结构进行访问，从而改变文档的结构，样式和内容。DOM 将文档解析为一个由节点和对象（包含属性和方法的对象）组成的结构集合。简言之，它会将 web 页面和脚本或程序语言连接起来。

节点的创建
// 用于创建一个指定 tag name 的 HTML 元素
document.creatElement();
// 新创建出的节点是“孤儿节点”，他并没有被挂载到 DOM 树上，无法看到它

// 需使用 appendChild() 或 insertBefore() 方法将孤儿节点插入到 DOM 树上
// 可以将孤儿节点挂载到它的内部，成为他的最后一个子节点
父节点.appendChild(孤儿节点);
// 可以将孤儿节点挂载到它的内部，成为它的“标杆子节点”之前的节点
父节点.insertBefore(孤儿节点, 标杆节点);
appendChild()
<body>
    <div id = "box">
        <p>段落1</p>
        <p>段落2</p>
        <p>段落3</p>
    </div>
    <script>
        var oBox = document.getElementById('box');
        // 创建孤儿子节点
        var oP = document.createElement('p');
        // 设置内部文字
        oP.innerText = '新来的';
        // 上树
        oBox.appendChild(oP);
    </script>
</body>

insertBefore()
<body>
    <div id = "box">
        <p>段落1</p>
        <p>段落2</p>
        <p>段落3</p>
    </div>
    <script>
        var oBox = document.getElementById('box');
        // 获取 oBox 下的所有 p 子节点
        var oPa = oBox.getElementsByTagName('p');
        // 创建孤儿子节点
        var oP = document.createElement('p');
        // 设置内部文字
        oP.innerText = '新来的';
        // 将 oP 插入到 oBox 中，插至 oPa[0] 之前的位置
        oBox.insertBefore(oP, oPa[0]);
    </script>
</body>
